def identity
  data = {}
  if !id.nil?
    data.merge!(to_h(:id))
  elsif !name.nil? && !domain.nil?
    data.merge!(to_h(:name))
    data.merge!({ :domain => @domain.identity })
  else
    raise CredentialsError, "#{self.class}: An Id, or a name with its domain, must be provided"
  end
  data
end
